const USERS_SHEET = "Users";
const STATUS_SHEET = "Statuses";
const LOCATIONS_SHEET = "Locations";

function getBoundSpreadsheet() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  if (!ss) throw new Error("找不到綁定的試算表");
  return ss;
}

function setup() {
  const ss = getBoundSpreadsheet();
  
  if (!ss.getSheetByName(USERS_SHEET)) {
    let uSheet = ss.insertSheet(USERS_SHEET);
    uSheet.appendRow(["id", "empId", "name", "isActive"]);
    for(let i=1; i<=12; i++) uSheet.appendRow(["emp_"+i, "E"+String(i).padStart(3,'0'), "人員 "+i, true]);
  }
  
  if (!ss.getSheetByName(STATUS_SHEET)) {
    let sSheet = ss.insertSheet(STATUS_SHEET);
    sSheet.appendRow(["docId", "userId", "date", "location", "customLocation", "updatedAt"]);
  }

  if (!ss.getSheetByName(LOCATIONS_SHEET)) {
    let lSheet = ss.insertSheet(LOCATIONS_SHEET);
    lSheet.appendRow(["id", "label", "sortOrder", "color"]);
    const defaultLocs = [
        { id: 'OCC', label: 'OCC (辦公室)', color: 'bg-green-100 text-green-800 border-green-200', sortOrder: 10 },
        { id: 'Home', label: '居家辦公', color: 'bg-blue-100 text-blue-800 border-blue-200', sortOrder: 20 },
        { id: 'Client', label: '外出開會/公差', color: 'bg-purple-100 text-purple-800 border-purple-200', sortOrder: 30 },
        { id: 'Leave', label: '請假', color: 'bg-red-100 text-red-800 border-red-200', sortOrder: 40 },
        { id: 'Daliao', label: '大寮機廠訓練教室', color: 'bg-teal-100 text-teal-800 border-teal-200', sortOrder: 50 },
        { id: 'South', label: '南機廠辦公室', color: 'bg-cyan-100 text-cyan-800 border-cyan-200', sortOrder: 60 },
        { id: 'Formosa', label: '美麗島', color: 'bg-pink-100 text-pink-800 border-pink-200', sortOrder: 70 },
        { id: 'A61', label: 'A61會議室', color: 'bg-indigo-100 text-indigo-800 border-indigo-200', sortOrder: 80 },

}

function doGet(e) {
  try {
    const ss = getBoundSpreadsheet();
    const res = { users: [], statuses: [], locations: [] };
    
    ['Users', 'Statuses', 'Locations'].forEach(sheetName => {
      const sheet = ss.getSheetByName(sheetName);
      if (sheet) {
        const data = sheet.getDataRange().getValues();
        if (data.length > 1) {
          const headers = data[0];
          for (let i = 1; i < data.length; i++) {
            let obj = {};
            for (let j = 0; j < headers.length; j++) obj[headers[j]] = data[i][j];
            res[sheetName.toLowerCase()].push(obj);
          }
        }
      }
    });
    return ContentService.createTextOutput(JSON.stringify(res)).setMimeType(ContentService.MimeType.JSON);
  } catch (error) { return ContentService.createTextOutput(JSON.stringify({ error: error.message })).setMimeType(ContentService.MimeType.JSON); }
}

function doPost(e) {
  try {
    const ss = getBoundSpreadsheet();
    let payloadStr = "";
    
    // 強化的防呆接收機制
    if (e.parameter && e.parameter.payload) {
      payloadStr = e.parameter.payload;
    } else if (e.postData && e.postData.contents) {
      payloadStr = e.postData.contents;
    }
    
    if(!payloadStr) throw new Error("No valid data received");
    
    const payload = JSON.parse(payloadStr);
    
    if (payload.action === "updateStatus") {
      const sheet = ss.getSheetByName(STATUS_SHEET);
      const data = payload.data;
      const docId = data.date + "_" + data.userId;
      const rows = sheet.getDataRange().getValues();
      let found = false;
      for (let i = 1; i < rows.length; i++) {
        if (rows[i][0] === docId) {
          sheet.getRange(i + 1, 2, 1, 5).setValues([[data.userId, data.date, data.location, data.customLocation, data.updatedAt]]);
          found = true; break;
        }
      }
      if (!found) sheet.appendRow([docId, data.userId, data.date, data.location, data.customLocation, data.updatedAt]);
    } else if (payload.action === "updateUser") {
      const sheet = ss.getSheetByName(USERS_SHEET);
      const data = payload.data;
      const rows = sheet.getDataRange().getValues();
      let found = false;
      for (let i = 1; i < rows.length; i++) {
        if (rows[i][0] === data.id) {
          sheet.getRange(i + 1, 2, 1, 3).setValues([[data.empId, data.name, data.isActive]]);
          found = true; break;
        }
      }
      if (!found) sheet.appendRow([data.id, data.empId, data.name, data.isActive]);
    } else if (payload.action === "updateLocation") {
      const sheet = ss.getSheetByName(LOCATIONS_SHEET);
      const data = payload.data;
      const rows = sheet.getDataRange().getValues();
      let found = false;
      for (let i = 1; i < rows.length; i++) {
        if (rows[i][0] === data.id) {
          sheet.getRange(i + 1, 2, 1, 3).setValues([[data.label, data.sortOrder, data.color]]);
          found = true; break;
        }
      }
      if (!found) sheet.appendRow([data.id, data.label, data.sortOrder, data.color]);
    } else if (payload.action === "deleteLocation") {
      const sheet = ss.getSheetByName(LOCATIONS_SHEET);
      const rows = sheet.getDataRange().getValues();
      for (let i = 1; i < rows.length; i++) {
        if (rows[i][0] === payload.id) { sheet.deleteRow(i + 1); break; }
      }
    }
    return ContentService.createTextOutput(JSON.stringify({ success: true })).setMimeType(ContentService.MimeType.JSON);
  } catch (error) { return ContentService.createTextOutput(JSON.stringify({ success: false, error: error.message })).setMimeType(ContentService.MimeType.JSON); }
}
